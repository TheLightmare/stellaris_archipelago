# ============================================================================
# Archipelago Bridge — Log-Based Outbound
# ============================================================================
# The mod communicates completed checks to the Python client by writing
# structured log lines to game.log. The client tails this file in real
# time and parses lines prefixed with "AP_".
#
# Log format:
#   AP_CHECK|<location_id>|<location_name>
#   AP_ENERGY_DEPOSIT|<amount>
#   AP_ENERGY_WITHDRAW|<amount>
#   AP_GOAL_COMPLETE
#   AP_HEARTBEAT|<year>
# ============================================================================

# --------------------------------------------------------------------------
# Heartbeat — confirms the mod is active (fires every month)
# --------------------------------------------------------------------------
ap_send_heartbeat = {
    log = "AP_HEARTBEAT"
}
